namespace vqf {

// 常量定义
constexpr double DEG2RAD = M_PI / 180.0;
constexpr double RAD2DEG = 180.0 / M_PI;
constexpr double EPSILON = 1e-10;

// VQF 参数配置
struct VqfParameters {
    // 时间常数
    double tauAcc = 3.0;           // 加速度计信任时间常数（秒）
    double tauMag = 0.0;           // 磁力计时间常数（0=禁用）
    
    // 静止检测参数
    bool restBiasEstEnabled = true;    // 启用静止零偏估计
    bool motionBiasEstEnabled = true;  // 启用运动零偏估计
    double restThGyr = 2.0;            // 静止检测陀螺阈值 (deg/s)
    double restThAcc = 0.5;            // 静止检测加速度阈值 (m/s²)
    double restMinT = 1.5;             // 最小静止时间 (秒)
    
    // 零偏估计参数
    double biasSigmaInit = 0.5;        // 初始零偏不确定性 (deg/s)
    double biasClip = 2.0;             // 零偏限制 (deg/s)
    
    // 内部参数
    double gyrTs = 0.01;               // 陀螺仪采样周期 (秒)
    double accTs = 0.01;               // 加速度计采样周期 (秒)
};

// 滤波器状态
struct VqfState {
    // 四元数状态 (w, x, y, z)
    Eigen::Quaterniond quat = Eigen::Quaterniond(1.0, 0.0, 0.0, 0.0);
    
    // 陀螺仪零偏估计
    Eigen::Vector3d bias = Eigen::Vector3d::Zero();
    
    // 零偏估计协方差矩阵 (3x3)
    Eigen::Matrix3d biasP = Eigen::Matrix3d::Identity() * 0.5 * 0.5;
    
    // 静止检测状态
    bool restDetected = false;
    double restT = 0.0;
    Eigen::Vector3d restLastGyrLp = Eigen::Vector3d::Zero();
    Eigen::Vector3d restLastAccLp = Eigen::Vector3d::Zero();
    std::array<double, 3> restGyrLpState = {0.0, 0.0, 0.0};
    std::array<double, 3> restAccLpState = {0.0, 0.0, 0.0};
    
    // 加速度计在低通滤波坐标系中的状态
    Eigen::Vector3d accLp = Eigen::Vector3d::Zero();
    std::array<double, 3> accLpState = {0.0, 0.0, 0.0};
    
    // 倾斜校正相关
    double kappa = 0.0;  // 倾斜校正增益
    
    // 运动零偏估计相关
    Eigen::Vector3d motionBiasEst = Eigen::Vector3d::Zero();
    bool motionBiasEstInitialized = false;
};

// 预计算系数
struct VqfCoeffs {
    double gyrTs;
    double accTs;
    
    // 静止检测低通滤波器系数
    double restGyrLpB[3];
    double restGyrLpA[3];
    double restAccLpB[3];
    double restAccLpA[3];
    
    // 加速度计低通滤波器系数（用于倾斜校正）
    double accLpB[3];
    double accLpA[3];
    
    // 零偏估计过程噪声
    double biasV;
    
    // 倾斜校正相关
    double tauAcc;
};

class VQF {
public:
    explicit VQF(const VqfParameters& params = VqfParameters());
    
    // 主要更新接口
    void update(const Eigen::Vector3d& gyro,    // rad/s
                const Eigen::Vector3d& accel,   // m/s²
                const Eigen::Vector3d* mag = nullptr);  // 可选磁力计 (μT)
    
    // 批量更新（离线处理）
    void updateBatch(const std::vector<Eigen::Vector3d>& gyroData,
                     const std::vector<Eigen::Vector3d>& accelData,
                     const std::vector<Eigen::Vector3d>* magData = nullptr);
    
    // 获取结果
    Eigen::Quaterniond getQuat() const { return state_.quat; }
    Eigen::Vector3d getBias() const { return state_.bias; }
    bool isRestDetected() const { return state_.restDetected; }
    
    // 获取 3D 模式结果（纯陀螺仪）
    Eigen::Quaterniond getQuat3D() const;
    
private:
    // 初始化系数
    void initCoeffs();
    
    // 预处理步骤
    void preprocessData(const Eigen::Vector3d& gyro, 
                        const Eigen::Vector3d& accel,
                        Eigen::Vector3d& gyroOut,
                        Eigen::Vector3d& accelOut);
    
    // 核心算法步骤
    void updateGyr(const Eigen::Vector3d& gyro);
    void updateAcc(const Eigen::Vector3d& accel);
    void updateMag(const Eigen::Vector3d& mag);
    
    // 静止检测
    void updateRest(const Eigen::Vector3d& gyro, const Eigen::Vector3d& accel);
    
    // 零偏估计
    void updateBias(const Eigen::Vector3d& gyro, const Eigen::Vector3d& accel);
    
    // 倾斜校正
    void updateTilt(const Eigen::Vector3d& accel);
    
    // 辅助函数
    static void filterVec(const double* in, int n, double tau, double ts,
                          const double* b, const double* a,
                          std::array<double, 3>& state,
                          Eigen::Vector3d& out);
    
    static Eigen::Matrix3d quatToRotMat(const Eigen::Quaterniond& q);
    static Eigen::Quaterniond rotVecToQuat(const Eigen::Vector3d& v);
    
    VqfParameters params_;
    VqfState state_;
    VqfCoeffs coeffs_;
    
    bool initialized_ = false;
};

} // namespace vqf

#endif // VQF_HPP

// 辅助函数：计算低通滤波器系数（2阶巴特沃斯）
static void designLowPass(double tau, double ts, double* b, double* a) {
    if (tau <= 0.0) {
        b[0] = 1.0; b[1] = 0.0; b[2] = 0.0;
        a[0] = 1.0; a[1] = 0.0; a[2] = 0.0;
        return;
    }
    
    double fc = 1.0 / (2.0 * M_PI * tau);  // 截止频率
    double C = std::tan(M_PI * fc * ts);
    double D = 2.0 * std::sqrt(2.0) * C + 4.0 * C * C + 2.0 * std::pow(2.0, 0.75) * std::sqrt(C) + 1.0;
    
    b[0] = 4.0 * C * C / D;
    b[1] = 2.0 * b[0];
    b[2] = b[0];
    
    a[0] = 1.0;
    a[1] = (2.0 - 8.0 * C * C) / D;
    a[2] = (1.0 - 2.0 * std::sqrt(2.0) * C + 4.0 * C * C 
            - 2.0 * std::pow(2.0, 0.75) * std::sqrt(C)) / D;
}

VQF::VQF(const VqfParameters& params) : params_(params) {
    initCoeffs();
}

void VQF::initCoeffs() {
    coeffs_.gyrTs = params_.gyrTs;
    coeffs_.accTs = params_.accTs;
    coeffs_.tauAcc = params_.tauAcc;
    
    // 设计静止检测低通滤波器
    double tauRest = 0.5;  // 静止检测滤波器时间常数
    designLowPass(tauRest, params_.gyrTs, coeffs_.restGyrLpB, coeffs_.restGyrLpA);
    designLowPass(tauRest, params_.accTs, coeffs_.restAccLpB, coeffs_.restAccLpA);
    
    // 设计加速度计低通滤波器（用于倾斜校正）
    designLowPass(params_.tauAcc, params_.accTs, coeffs_.accLpB, coeffs_.accLpA);
    
    // 零偏估计过程噪声
    double biasSigma0 = params_.biasSigmaInit * DEG2RAD;
    coeffs_.biasV = (biasSigma0 * biasSigma0) * params_.gyrTs;
}

void VQF::preprocessData(const Eigen::Vector3d& gyro,
                         const Eigen::Vector3d& accel,
                         Eigen::Vector3d& gyroOut,
                         Eigen::Vector3d& accelOut) {
    // ============================================
    // 关键预处理步骤 1: 单位转换和范围检查
    // ============================================
    
    // 确保陀螺仪单位是 rad/s（如果输入是 deg/s，需要转换）
    // Gyroflow 通常接收 rad/s，这里假设输入已经是 rad/s
    
    // 检查加速度计模长，确保合理
    double accNorm = accel.norm();
    if (accNorm < 0.5 || accNorm > 20.0) {
        // 加速度异常，使用上一次的值或预测值
        accelOut = state_.quat.inverse() * Eigen::Vector3d(0, 0, 9.81);
    } else {
        accelOut = accel;
    }
    
    gyroOut = gyro;
    
    // ============================================
    // 关键预处理步骤 2: 零偏初始化（第一次运行）
    // ============================================
    if (!initialized_) {
        // 使用初始加速度计读数初始化姿态
        // 假设初始时刻设备静止，加速度计指向重力方向
        Eigen::Vector3d g = accelOut.normalized();
        
        // 计算初始 roll 和 pitch
        double roll = std::atan2(-g.y(), -g.z());
        double pitch = std::atan2(g.x(), std::sqrt(g.y() * g.y() + g.z() * g.z()));
        
        // 构建初始四元数（假设 yaw = 0）
        Eigen::AngleAxisd rollAngle(roll, Eigen::Vector3d::UnitX());
        Eigen::AngleAxisd pitchAngle(pitch, Eigen::Vector3d::UnitY());
        state_.quat = rollAngle * pitchAngle;
        
        initialized_ = true;
    }
}

void VQF::update(const Eigen::Vector3d& gyro,
                 const Eigen::Vector3d& accel,
                 const Eigen::Vector3d* mag) {
    // 预处理
    Eigen::Vector3d gyroProc, accelProc;
    preprocessData(gyro, accel, gyroProc, accelProc);
    
    // 步骤 1: 静止检测更新（必须在陀螺更新前）
    updateRest(gyroProc, accelProc);
    
    // 步骤 2: 陀螺仪积分（使用去零偏后的数据）
    updateGyr(gyroProc);
    
    // 步骤 3: 加速度计倾斜校正
    updateAcc(accelProc);
    
    // 步骤 4: 磁力计更新（如果提供）
    if (mag != nullptr) {
        updateMag(*mag);
    }
    
    // 步骤 5: 零偏估计更新
    updateBias(gyroProc, accelProc);
}

void VQF::updateRest(const Eigen::Vector3d& gyro, const Eigen::Vector3d& accel) {
    if (!params_.restBiasEstEnabled) {
        state_.restDetected = false;
        return;
    }
    
    // 对陀螺仪数据进行低通滤波
    double gyrIn[3] = {gyro.x(), gyro.y(), gyro.z()};
    double gyrOut[3];
    filterVec(gyrIn, 3, 0.5, params_.gyrTs,
              coeffs_.restGyrLpB, coeffs_.restGyrLpA,
              state_.restGyrLpState, state_.restLastGyrLp);
    
    // 计算平方偏差
    double dev = std::pow(gyro.x() - state_.restLastGyrLp.x(), 2) +
                 std::pow(gyro.y() - state_.restLastGyrLp.y(), 2) +
                 std::pow(gyro.z() - state_.restLastGyrLp.z(), 2);
    
    // 阈值判断（转换为 rad/s）
    double thGyr = std::pow(params_.restThGyr * DEG2RAD, 2);
    
    if (dev < thGyr) {
        state_.restT += params_.gyrTs;
        if (state_.restT >= params_.restMinT) {
            state_.restDetected = true;
        }
    } else {
        state_.restT = 0.0;
        state_.restDetected = false;
    }
}

void VQF::updateGyr(const Eigen::Vector3d& gyro) {
    // 去零偏
    Eigen::Vector3d gyroUnbiased = gyro - state_.bias;
    
    // 四元数积分（一阶近似）
    Eigen::Vector3d angle = gyroUnbiased * params_.gyrTs;
    double angleNorm = angle.norm();
    
    Eigen::Quaterniond deltaQ;
    if (angleNorm > EPSILON) {
        Eigen::Vector3d axis = angle / angleNorm;
        deltaQ = Eigen::Quaterniond(std::cos(angleNorm / 2.0),
                                    axis.x() * std::sin(angleNorm / 2.0),
                                    axis.y() * std::sin(angleNorm / 2.0),
                                    axis.z() * std::sin(angleNorm / 2.0));
    } else {
        deltaQ = Eigen::Quaterniond(1.0, 0.5 * angle.x(), 
                                    0.5 * angle.y(), 0.5 * angle.z());
        deltaQ.normalize();
    }
    
    // 更新姿态
    state_.quat = (state_.quat * deltaQ).normalized();
}

void VQF::updateAcc(const Eigen::Vector3d& accel) {
    // 将加速度转换到世界坐标系
    Eigen::Vector3d accWorld = state_.quat * accel;
    
    // 低通滤波获取重力方向
    double accIn[3] = {accel.x(), accel.y(), accel.z()};
    Eigen::Vector3d accLp;
    filterVec(accIn, 3, params_.tauAcc, params_.accTs,
              coeffs_.accLpB, coeffs_.accLpA,
              state_.accLpState, accLp);
    
    // 计算倾斜误差（叉乘）
    Eigen::Vector3d gRef(0, 0, 1.0);  // 参考重力方向
    Eigen::Vector3d gEst = (state_.quat * accLp).normalized();
    
    // 只有当加速度接近重力时才进行校正
    double accNorm = accel.norm();
    if (std::abs(accNorm - 9.81) > 2.0) {
        return;  // 线性加速度过大，跳过校正
    }
    
    // 计算倾斜误差向量
    Eigen::Vector3d tiltError = gEst.cross(gRef);
    
    // 计算自适应增益 kappa
    double k = (2.0 / 9.81) * tiltError.norm();
    k = std::min(k, 1.0);  // 限制最大增益
    
    // 倾斜校正（使用 SLERP 近似）
    if (tiltError.norm() > EPSILON) {
        Eigen::Vector3d correctionAxis = tiltError.normalized();
        double correctionAngle = k * tiltError.norm() * params_.accTs / params_.tauAcc;
        
        Eigen::Quaterniond correctionQ(
            std::cos(correctionAngle / 2.0),
            correctionAxis.x() * std::sin(correctionAngle / 2.0),
            correctionAxis.y() * std::sin(correctionAngle / 2.0),
            correctionAxis.z() * std::sin(correctionAngle / 2.0)
        );
        
        state_.quat = (correctionQ * state_.quat).normalized();
    }
}

void VQF::updateBias(const Eigen::Vector3d& gyro, const Eigen::Vector3d& accel) {
    if (!params_.motionBiasEstEnabled && !params_.restBiasEstEnabled) {
        return;
    }
    
    // 静止状态下的零偏估计
    if (state_.restDetected && params_.restBiasEstEnabled) {
        // 简单平均法
        double alpha = 0.01;  // 平滑因子
        state_.bias = (1.0 - alpha) * state_.bias + alpha * gyro;
        
        // 限制零偏范围
        double biasLimit = params_.biasClip * DEG2RAD;
        state_.bias = state_.bias.cwiseMax(-biasLimit).cwiseMin(biasLimit);
        return;
    }
    
    // 运动状态下的零偏估计（卡尔曼滤波简化版）
    if (params_.motionBiasEstEnabled && !state_.restDetected) {
        // 预测步骤：协方差增加
        state_.biasP += Eigen::Matrix3d::Identity() * coeffs_.biasV;
        
        // 计算创新（基于加速度计一致性检查）
        Eigen::Vector3d accWorld = state_.quat * accel;
        Eigen::Vector3d gRef(0, 0, 9.81);
        Eigen::Vector3d innovation = accWorld - gRef;
        
        // 简化的卡尔曼增益计算
        double R = 0.1;  // 测量噪声
        Eigen::Matrix3d K = state_.biasP * (state_.biasP + Eigen::Matrix3d::Identity() * R).inverse();
        
        // 假设创新主要由陀螺仪零偏引起
        Eigen::Vector3d biasCorrection = K * innovation;
        state_.bias += biasCorrection * 0.1;  // 缩放因子
        
        // 更新协方差
        state_.biasP = (Eigen::Matrix3d::Identity() - K) * state_.biasP;
        
        // 限制零偏
        double biasLimit = params_.biasClip * DEG2RAD;
        state_.bias = state_.bias.cwiseMax(-biasLimit).cwiseMin(biasLimit);
    }
}

void VQF::updateMag(const Eigen::Vector3d& mag) {
    // 注意：oxkitsune/vqf 不支持磁力计，这里预留接口
    // 如果需要完整 9D 实现，需要参考官方 VQF C++ 代码
    (void)mag;  // 避免未使用警告
}

void VQF::filterVec(const double* in, int n, double tau, double ts,
                    const double* b, const double* a,
                    std::array<double, 3>& state,
                    Eigen::Vector3d& out) {
    // 2阶IIR滤波器实现
    for (int i = 0; i < n; ++i) {
        double x = in[i];
        double y = b[0] * x + state[0];
        state[0] = b[1] * x - a[1] * y + state[1];
        state[1] = b[2] * x - a[2] * y;
        
        if (i == 0) out.x() = y;
        else if (i == 1) out.y() = y;
        else if (i == 2) out.z() = y;
    }
}

void VQF::updateBatch(const std::vector<Eigen::Vector3d>& gyroData,
                      const std::vector<Eigen::Vector3d>& accelData,
                      const std::vector<Eigen::Vector3d>* magData) {
    size_t N = gyroData.size();
    if (accelData.size() != N) {
        throw std::invalid_argument("Gyro and accel data size mismatch");
    }
    
    for (size_t i = 0; i < N; ++i) {
        if (magData != nullptr && !magData->empty()) {
            update(gyroData[i], accelData[i], &(*magData)[i]);
        } else {
            update(gyroData[i], accelData[i], nullptr);
        }
    }
}

Eigen::Quaterniond VQF::getQuat3D() const {
    // 返回纯陀螺仪积分结果（无加速度计校正）
    // 这里简化处理，实际应该保存纯积分状态
    return state_.quat;
}

// 1. 单位检查与转换
//    - 陀螺仪: 确保是 rad/s（Gyroflow 内部使用 rad/s）
//    - 加速度计: 确保是 m/s²

// 2. 异常值处理
//    - 检查加速度计模长是否在合理范围 (0.5g - 2g)
//    - 如果异常，使用预测值替代

// 3. 初始姿态估计
//    - 使用初始加速度计读数计算 roll 和 pitch
//    - 假设初始静止，重力方向为参考

// 4. 坐标系对齐（在 Gyroflow 上层完成）
//    - 根据设备类型（GoPro、Sony 等）进行坐标转换
//    - 通过 IMU 方向字符串（如 "XYZ"、"YXZ"）调整轴顺序
#include "vqf.hpp"
#include <iostream>
#include <vector>

int main() {
    using namespace vqf;
    
    // 配置参数（匹配 Gyroflow 默认值）
    VqfParameters params;
    params.tauAcc = 3.0;           // 加速度计信任时间
    params.restBiasEstEnabled = true;
    params.motionBiasEstEnabled = true;
    params.restThGyr = 2.0;        // 2 deg/s 静止阈值
    params.biasClip = 2.0;         // 2 deg/s 零偏限制
    params.gyrTs = 0.01;           // 100Hz
    params.accTs = 0.01;
    
    // 创建 VQF 实例
    VQF vqf(params);
    
    // 模拟数据
    std::vector<Eigen::Vector3d> gyroData;
    std::vector<Eigen::Vector3d> accelData;
    
    // 填充数据...
    for (int i = 0; i < 1000; ++i) {
        gyroData.push_back(Eigen::Vector3d(0.01, 0.02, 0.001));  // rad/s
        accelData.push_back(Eigen::Vector3d(0.0, 0.0, 9.81));    // m/s²
    }
    
    // 批量处理
    vqf.updateBatch(gyroData, accelData);
    
    // 获取结果
    auto quat = vqf.getQuat();
    std::cout << "Final quaternion: " << quat.w() << ", " 
              << quat.x() << ", " << quat.y() << ", " << quat.z() << std::endl;
			  
			//  预处理	在 Rust 层完成	在 preprocessData 中
//坐标转换	在上层完成	需要用户预处理
    
    return 0;
}
