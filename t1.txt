#include <iostream>
#include <fstream>
#include <vector>

int main() {
    using namespace vqf;
    
    VqfParameters params;
    params.tauAcc = 3.0;
    params.restBiasEstEnabled = true;
    params.motionBiasEstEnabled = true;
    
    VQF vqf(params);
    
    // 从 Gyroflow 导出的 CSV 数据读取
    // 格式: timestamp, gyro_x, gyro_y, gyro_z, acc_x, acc_y, acc_z
    std::vector<ImuSample> samples;
    
    std::ifstream file("gyroflow_data.csv");
    double ts, gx, gy, gz, ax, ay, az;
    while (file >> ts >> gx >> gy >> gz >> ax >> ay >> az) {
        ImuSample sample;
        sample.timestamp = ts;  // 使用 Gyroflow 导出的实际时间戳
        sample.gyro = Eigen::Vector3d(gx, gy, gz);    // rad/s
        sample.accel = Eigen::Vector3d(ax, ay, az);   // m/s²
        samples.push_back(sample);
    }
    
    // 批量处理（自动计算 dt）
    vqf.updateBatch(samples);
    
    // 或者逐样本处理（适用于实时流）
    // for (const auto& s : samples) {
    //     vqf.update(s);
    // }
    
    auto quat = vqf.getQuat();
    std::cout << "Final quaternion: " << quat.w() << ", " 
              << quat.x() << ", " << quat.y() << ", " << quat.z() << std::endl;
    
    return 0;
}

// ========== 关键修正：使用实际时间戳的更新 ==========

void VQF::update(const ImuSample& sample) {
    // 第一次样本：初始化时间戳，不计算 dt
    if (firstSample_) {
        lastTimestamp_ = sample.timestamp;
        firstSample_ = false;
        
        // 初始化姿态（使用加速度计）
        Eigen::Vector3d g = sample.accel.normalized();
        double roll = std::atan2(-g.y(), -g.z());
        double pitch = std::atan2(g.x(), std::sqrt(g.y() * g.y() + g.z() * g.z()));
        
        Eigen::AngleAxisd rollAngle(roll, Eigen::Vector3d::UnitX());
        Eigen::AngleAxisd pitchAngle(pitch, Eigen::Vector3d::UnitY());
        state_.quat = rollAngle * pitchAngle;
        
        return;
    }
    
    // 计算实际时间差 dt（秒）
    double dt = sample.timestamp - lastTimestamp_;
    
    // 异常时间戳处理
    if (dt <= 0.0 || dt > 1.0) {
        // 时间戳异常（回退或跳变），使用默认采样率
        std::cerr << "Warning: Invalid dt " << dt << "s, using default 0.01s" << std::endl;
        dt = 0.01;
    }
    
    // 更新滤波器系数（如果 dt 变化显著）
    if (std::abs(dt - cachedGyrTs_) > 0.001) {
        updateCoeffsForDt(dt);
        cachedGyrTs_ = dt;
    }
    
    // 执行更新
    updateInternal(sample.gyro, sample.accel, dt, 
                   sample.hasMag ? &sample.mag : nullptr);
    
    // 更新时间戳
    lastTimestamp_ = sample.timestamp;
}

void VQF::updateInternal(const Eigen::Vector3d& gyro,
                         const Eigen::Vector3d& accel,
                         double dt,
                         const Eigen::Vector3d* mag) {
    // 预处理
    Eigen::Vector3d gyroProc, accelProc;
    preprocessData(gyro, accel, dt, gyroProc, accelProc);
    
    // 步骤 1: 静止检测（使用实际 dt）
    updateRest(gyroProc, accelProc, dt);
    
    // 步骤 2: 陀螺仪积分（使用实际 dt）
    updateGyr(gyroProc, dt);
    
    // 步骤 3: 加速度计倾斜校正（使用实际 dt）
    updateAcc(accelProc, dt);
    
    // 步骤 4: 磁力计更新（如果提供）
    if (mag != nullptr) {
        updateMag(*mag, dt);
    }
    
    // 步骤 5: 零偏估计更新（使用实际 dt）
    updateBias(gyroProc, accelProc, dt);
}

// ========== 关键：根据实际 dt 动态更新系数 ==========

void VQF::updateCoeffsForDt(double dt) {
    // 重新设计低通滤波器系数（基于实际 dt）
    double tauRest = 0.5;
    designLowPass(tauRest, dt, coeffs_.restGyrLpB, coeffs_.restGyrLpA);
    designLowPass(tauRest, dt, coeffs_.restAccLpB, coeffs_.restAccLpA);
    designLowPass(params_.tauAcc, dt, coeffs_.accLpB, coeffs_.accLpA);
    
    // 更新过程噪声（与 dt 成正比）
    double biasSigma0 = params_.biasSigmaInit * DEG2RAD;
    coeffs_.biasV = (biasSigma0 * biasSigma0) * dt;
    
    coeffs_.gyrTs = dt;
    coeffs_.accTs = dt;
}

// ========== 关键修正：所有更新函数使用实际 dt ==========

void VQF::updateGyr(const Eigen::Vector3d& gyro, double dt) {
    // 去零偏
    Eigen::Vector3d gyroUnbiased = gyro - state_.bias;
    
    // 四元数积分（使用实际 dt）
    Eigen::Vector3d angle = gyroUnbiased * dt;  // 关键：使用实际 dt
    double angleNorm = angle.norm();
    
    Eigen::Quaterniond deltaQ;
    if (angleNorm > EPSILON) {
        Eigen::Vector3d axis = angle / angleNorm;
        deltaQ = Eigen::Quaterniond(std::cos(angleNorm / 2.0),
                                    axis.x() * std::sin(angleNorm / 2.0),
                                    axis.y() * std::sin(angleNorm / 2.0),
                                    axis.z() * std::sin(angleNorm / 2.0));
    } else {
        // 小角度近似
        deltaQ = Eigen::Quaterniond(1.0, 
                                    0.5 * angle.x(), 
                                    0.5 * angle.y(), 
                                    0.5 * angle.z());
        deltaQ.normalize();
    }
    
    state_.quat = (state_.quat * deltaQ).normalized();
}

void VQF::updateRest(const Eigen::Vector3d& gyro, 
                     const Eigen::Vector3d& accel, 
                     double dt) {
    if (!params_.restBiasEstEnabled) {
        state_.restDetected = false;
        return;
    }
    
    // 使用实际 dt 进行低通滤波
    double gyrIn[3] = {gyro.x(), gyro.y(), gyro.z()};
    filterVec(gyrIn, 3, 0.5, dt,  // 关键：使用实际 dt
              coeffs_.restGyrLpB, coeffs_.restGyrLpA,
              state_.restGyrLpState, state_.restLastGyrLp);
    
    // 计算平方偏差
    double dev = std::pow(gyro.x() - state_.restLastGyrLp.x(), 2) +
                 std::pow(gyro.y() - state_.restLastGyrLp.y(), 2) +
                 std::pow(gyro.z() - state_.restLastGyrLp.z(), 2);
    
    double thGyr = std::pow(params_.restThGyr * DEG2RAD, 2);
    
    // 使用实际 dt 累积静止时间
    if (dev < thGyr) {
        state_.restT += dt;  // 关键：使用实际 dt
        if (state_.restT >= params_.restMinT) {
            state_.restDetected = true;
        }
    } else {
        state_.restT = 0.0;
        state_.restDetected = false;
    }
}

void VQF::updateAcc(const Eigen::Vector3d& accel, double dt) {
    // 将加速度转换到世界坐标系
    Eigen::Vector3d accWorld = state_.quat * accel;
    
    // 使用实际 dt 进行低通滤波
    double accIn[3] = {accel.x(), accel.y(), accel.z()};
    Eigen::Vector3d accLp;
    filterVec(accIn, 3, params_.tauAcc, dt,  // 关键：使用实际 dt
              coeffs_.accLpB, coeffs_.accLpA,
              state_.accLpState, accLp);
    
    // 检查加速度合理性
    double accNorm = accel.norm();
    if (std::abs(accNorm - 9.81) > 2.0) {
        return;  // 线性加速度过大
    }
    
    // 计算倾斜误差
    Eigen::Vector3d gRef(0, 0, 1.0);
    Eigen::Vector3d gEst = (state_.quat * accLp).normalized();
    Eigen::Vector3d tiltError = gEst.cross(gRef);
    
    // 自适应增益（与 dt 无关，但校正步长与 dt 相关）
    double k = (2.0 / 9.81) * tiltError.norm();
    k = std::min(k, 1.0);
    
    // 倾斜校正（校正角度与 dt 成正比）
    if (tiltError.norm() > EPSILON) {
        Eigen::Vector3d correctionAxis = tiltError.normalized();
        // 关键：校正角度使用实际 dt
        double correctionAngle = k * tiltError.norm() * dt / params_.tauAcc;
        
        Eigen::Quaterniond correctionQ(
            std::cos(correctionAngle / 2.0),
            correctionAxis.x() * std::sin(correctionAngle / 2.0),
            correctionAxis.y() * std::sin(correctionAngle / 2.0),
            correctionAxis.z() * std::sin(correctionAngle / 2.0)
        );
        
        state_.quat = (correctionQ * state_.quat).normalized();
    }
}

void VQF::updateBias(const Eigen::Vector3d& gyro, 
                     const Eigen::Vector3d& accel, 
                     double dt) {
    if (!params_.motionBiasEstEnabled && !params_.restBiasEstEnabled) {
        return;
    }
    
    // 静止状态下的零偏估计
    if (state_.restDetected && params_.restBiasEstEnabled) {
        // 指数移动平均（时间常数与 dt 相关）
        double alpha = 1.0 - std::exp(-dt / 1.0);  // 1秒时间常数
        state_.bias = (1.0 - alpha) * state_.bias + alpha * gyro;
        
        double biasLimit = params_.biasClip * DEG2RAD;
        state_.bias = state_.bias.cwiseMax(-biasLimit).cwiseMin(biasLimit);
        return;
    }
    
    // 运动状态下的零偏估计
    if (params_.motionBiasEstEnabled && !state_.restDetected) {
        // 预测步骤：协方差增加（与 dt 成正比）
        state_.biasP += Eigen::Matrix3d::Identity() * coeffs_.biasV * (dt / 0.01);
        
        // ... 其余卡尔曼滤波步骤 ...
    }
}

// 滤波器函数也需要使用实际 dt
void VQF::filterVec(const double* in, int n, double tau, double dt,
                    const double* b, const double* a,
                    std::array<double, 3>& state,
                    Eigen::Vector3d& out) {
    // 使用实际 dt 的滤波器实现
    for (int i = 0; i < n; ++i) {
        double x = in[i];
        double y = b[0] * x + state[0];
        state[0] = b[1] * x - a[1] * y + state[1];
        state[1] = b[2] * x - a[2] * y;
        
        if (i == 0) out.x() = y;
        else if (i == 1) out.y() = y;
        else if (i == 2) out.z() = y;
    }
}

// 批量更新实现
void VQF::updateBatch(const std::vector<ImuSample>& samples) {
    for (const auto& sample : samples) {
        update(sample);
    }
}

// 兼容旧接口
void VQF::updateFixedDt(const Eigen::Vector3d& gyro,
                        const Eigen::Vector3d& accel,
                        double dt,
                        const Eigen::Vector3d* mag) {
    if (firstSample_) {
        firstSample_ = false;
        lastTimestamp_ = 0.0;
        cachedGyrTs_ = dt;
        updateCoeffsForDt(dt);
    }
    
    updateInternal(gyro, accel, dt, mag);
    lastTimestamp_ += dt;
}

#ifndef VQF_HPP

// ... 其他常量定义保持不变 ...

// IMU 数据样本（包含时间戳）
struct ImuSample {
    double timestamp;  // 秒（Unix 时间或相对时间）
    Eigen::Vector3d gyro;   // rad/s
    Eigen::Vector3d accel;  // m/s²
    Eigen::Vector3d mag;    // μT（可选，默认为零）
    bool hasMag = false;
};

// VQF 参数配置（保持不变）
struct VqfParameters {
    double tauAcc = 3.0;
    double tauMag = 0.0;
    bool restBiasEstEnabled = true;
    bool motionBiasEstEnabled = true;
    double restThGyr = 2.0;
    double restThAcc = 0.5;
    double restMinT = 1.5;
    double biasSigmaInit = 0.5;
    double biasClip = 2.0;
    // 移除固定的 gyrTs 和 accTs，改为动态计算
};

// ... VqfState 和 VqfCoeffs 保持不变 ...

class VQF {
public:
    explicit VQF(const VqfParameters& params = VqfParameters());
    
    // ========== 关键修正：使用实际时间戳的更新接口 ==========
    
    // 单样本更新（带时间戳）
    void update(const ImuSample& sample);
    
    // 批量更新（时间戳数组）
    void updateBatch(const std::vector<ImuSample>& samples);
    
    // 兼容旧接口：固定时间步长更新（不推荐）
    void updateFixedDt(const Eigen::Vector3d& gyro,
                       const Eigen::Vector3d& accel,
                       double dt,  // 实际时间差（秒）
                       const Eigen::Vector3d* mag = nullptr);
    
    // 获取结果（保持不变）
    Eigen::Quaterniond getQuat() const { return state_.quat; }
    Eigen::Vector3d getBias() const { return state_.bias; }
    bool isRestDetected() const { return state_.restDetected; }
    double getLastTimestamp() const { return lastTimestamp_; }
    
private:
    // 核心更新步骤（使用实际 dt）
    void updateInternal(const Eigen::Vector3d& gyro,
                        const Eigen::Vector3d& accel,
                        double dt,  // 实际计算的时间差
                        const Eigen::Vector3d* mag);
    
    // 根据实际 dt 重新计算滤波器系数
    void updateCoeffsForDt(double dt);
    
    // 预处理步骤（现在接收 dt 参数）
    void preprocessData(const Eigen::Vector3d& gyro, 
                        const Eigen::Vector3d& accel,
                        double dt,
                        Eigen::Vector3d& gyroOut,
                        Eigen::Vector3d& accelOut);
    
    // 其他内部函数（保持不变）
    void updateGyr(const Eigen::Vector3d& gyro, double dt);
    void updateAcc(const Eigen::Vector3d& accel, double dt);
    void updateMag(const Eigen::Vector3d& mag, double dt);
    void updateRest(const Eigen::Vector3d& gyro, const Eigen::Vector3d& accel, double dt);
    void updateBias(const Eigen::Vector3d& gyro, const Eigen::Vector3d& accel, double dt);
    void updateTilt(const Eigen::Vector3d& accel, double dt);
    
    static void filterVec(const double* in, int n, double tau, double dt,
                          const double* b, const double* a,
                          std::array<double, 3>& state,
                          Eigen::Vector3d& out);
    
    // ... 其他辅助函数 ...
    
    VqfParameters params_;
    VqfState state_;
    VqfCoeffs coeffs_;
    
    // 新增：时间戳管理
    double lastTimestamp_ = -1.0;
    bool firstSample_ = true;
    
    // 缓存的系数（用于不同 dt）
    double cachedGyrTs_ = 0.01;
    bool initialized_ = false;
};

} // namespace vqf
